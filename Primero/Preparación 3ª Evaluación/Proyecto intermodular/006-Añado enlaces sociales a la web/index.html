// el mini-router PHP

$page = $_GET['page'] ?? 'home';

switch ($page) {
  case 'aviso-legal':
    include 'partials/aviso.php';        break;
  case 'politica-privacidad':
    include 'partials/privacidad.php';   break;
  case 'politica-cookies':
    include 'partials/cookies.php';      break;
  default:
    include 'partials/home.php';
}

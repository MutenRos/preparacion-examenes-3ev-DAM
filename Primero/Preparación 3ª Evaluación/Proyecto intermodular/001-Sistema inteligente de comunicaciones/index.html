# Resumen async de cada email cuando el usuario lo abre
@app.route("/api/summary/<int:msg_id>")
def api_summary(msg_id):
    body = fetch_message_body(msg_id)
    payload = {
        "model":  OLLAMA_MODEL,
        "prompt": f"Resume en 2 frases:\n\n{body[:4000]}",
        "stream": False,
    }
    r = requests.post(OLLAMA_URL, json=payload, timeout=90)
    return jsonify({"summary": r.json()["response"]})

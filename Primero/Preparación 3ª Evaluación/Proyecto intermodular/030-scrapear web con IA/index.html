# 005-afinar un poco el prompt.py import requests import subprocess url = "https://jocarsa.com" response = requests.get(url) html = response.text prompt = f"Te voy a pasar el codigo... {html}" result = subprocess.run( ["ollama", "run", "qwen2.5-coder:7b", prompt], capture_output=True, text=True ) print("RESUMEN:\n") print(result.stdout)

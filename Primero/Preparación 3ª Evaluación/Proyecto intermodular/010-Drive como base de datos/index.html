📜 PHP lee Google Sheets como si fuera un fichero local — tienda.php
// La URL del Google Sheet publicado como CSV
$url = "https://docs.google.com/spreadsheets/d/e/2PACX-.../pub?output=csv";

if (($h = fopen($url, "r")) !== false) {
    $headers = fgetcsv($h);     // 1ª fila = nombres columna
    while (($row = fgetcsv($h, 1000, ",")) !== false) {
        $productos[] = array_combine($headers, $row);
    }
    fclose($h);
}
// → tienda en tiempo real, las imágenes vienen de Google Drive

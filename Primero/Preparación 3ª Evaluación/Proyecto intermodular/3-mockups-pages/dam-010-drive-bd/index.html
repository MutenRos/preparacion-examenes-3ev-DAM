🐘 PHP — leer el CSV remoto 003-array nombrado.php

// Descarga el Sheet publicado y lo convierte en array asociativo $url = "https://docs.google.com/spreadsheets/d/e/.../pub?output=csv"; $datos = []; if (($h = fopen($url, "r")) !== false) { $cab = fgetcsv($h, 1000, ","); // cabeceras while (($fila = fgetcsv($h, 1000, ",")) !== false) { $datos[] = array_combine($cab, $fila); } fclose($h); } // $datos -> [['id'=>'1','nombre'=>'Balón',...], ...] foreach ($datos as $p) { echo "<article>" . "<h3>" . $p["nombre"] . "</h3>" . "<p>" . $p["precio"] . " €</p>" . "</article>"; }

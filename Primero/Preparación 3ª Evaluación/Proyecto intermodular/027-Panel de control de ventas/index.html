// el corazón del CRUD genérico — describe.php
// Lee la estructura de cualquier tabla y construye formulario
function describe_table(PDO $pdo, $tabla): array {
    $stmt = $pdo->query("PRAGMA table_info('" . $tabla . "')");
    return $stmt->fetchAll(PDO::FETCH_ASSOC);
}

// Genera input según el tipo de cada columna
foreach (describe_table($pdo, $tabla) as $col) {
    switch ($col['type']) {
        case 'INTEGER': input('number', $col['name']); break;
        case 'TEXT':    input('text',   $col['name']); break;
        case 'REAL':    input('number', $col['name'], step='0.01');
    }
}

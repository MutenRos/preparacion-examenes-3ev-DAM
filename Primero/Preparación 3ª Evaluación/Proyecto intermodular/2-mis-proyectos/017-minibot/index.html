normalize_url
def normalize_url(url: str) -> str:
    p = urlparse(url)
    return urlunparse(p._replace(
        scheme = p.scheme.lower(),
        netloc = p.netloc.lower(),
        fragment = "",    # sin #anchors
        path = p.path.rstrip("/") or "/"
    ))

Evita visitar la misma página dos veces con distinto formato de URL (mayúsculas, slash final, fragmento).

BFS loop central
queue = deque([START_URL])
visited = set()

while queue:
    url = queue.popleft()
    if url in visited: continue
    visited.add(url)

    r = requests.get(url, timeout=15)
    soup = BeautifulSoup(r.text, 'html.parser')
    title = soup.title.string
    save_to_db(url, title)

    for a in soup.find_all('a', href=True):
        abs_url = urljoin(url, a['href'])
        if same_domain(abs_url):
            queue.append(normalize_url(abs_url))

    time.sleep(DELAY_SECONDS)  # 1 seg cortesía

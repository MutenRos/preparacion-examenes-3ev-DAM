// 024-animar solo al hablar.php — el corazón del lip-sync
// Morph targets sincronizados con la síntesis de voz
const utterance = new SpeechSynthesisUtterance(text);

utterance.onstart = () => {
    morphInterval = setInterval(() => {
        mesh.morphTargetInfluences[0] =
            Math.sin(Date.now() / 100) * 0.5 + 0.5;
    }, 50);
};

utterance.onend = () => {
    clearInterval(morphInterval);
    mesh.morphTargetInfluences[0] = 0;   // boca cerrada
};

speechSynthesis.speak(utterance);
